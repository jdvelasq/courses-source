Paneles de DataFrames¶

[17]:

## se crean los DataFrames
df1 = pd.DataFrame({
    'colA': [1, 2],
    'colB': [3, 4]})

df2 = pd.DataFrame({
    'colB': [5, 6],
    'colC': [7, 8]})

df3 = pd.DataFrame({
    'colC': [9, 0],
    'colD': [1, 2]})

[18]:

## creación del panel como un diccionario
pdPanel = { 'df1': df1,
            'df2': df2,
            'df3': df3}
print(pdPanel)
