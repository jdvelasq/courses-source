Especificación del modelo¶

[6]:

#
#  Configuración de las capas
#
model = tf.keras.models.Sequential(
    [
        tf.keras.layers.Conv2D(32, (3, 3), activation="relu", input_shape=(32, 32, 3)),
        tf.keras.layers.MaxPooling2D((2, 2)),
        tf.keras.layers.Conv2D(64, (3, 3), activation="relu"),
        tf.keras.layers.MaxPooling2D((2, 2)),
        tf.keras.layers.Conv2D(64, (3, 3), activation="relu"),
        #
        #  Adiciona una capa para ejecutar la clasificación.
        #  La capa Flattern convierte la salida 3D de la capa anterior
        #  en 1D. Adiciona una capa y luego usa una capa de 10
        #  neuronas (10 clases) con activación softmax
        #
        tf.keras.layers.Flatten(),
        tf.keras.layers.Dense(64, activation='relu'),
        tf.keras.layers.Dense(10),
    ]
)

#
# Características del modelo creado
#
model.summary()

Compilación y entrenamiento del modelo¶

[8]:

model.compile(
    optimizer="adam",
    loss=tf.keras.losses.SparseCategoricalCrossentropy(from_logits=True),
    metrics=["accuracy"],
)

history = model.fit(
    train_images, train_labels, epochs=10, validation_data=(test_images, test_labels)
)
